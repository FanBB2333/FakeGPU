2 · Switch GPU profiles

Hardware compatibility 
$ FAKEGPU_TERMINAL_REPORT=0 python3 - <<'PY'
import fakegpu
fakegpu.init(runtime="fakecuda",
  devices="test-512m,rtx2080ti,rtx5090")
import torch
for i in range(torch.cuda.device_count()):
    p = torch.cuda.get_device_properties(i)
    print(f"GPU {i}: {p.name} | "
          f"{p.total_memory / 1024**3:.1f} GiB | "
          f"cc {p.major}.{p.minor}")
